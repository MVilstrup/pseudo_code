\begin{algorithm}
\caption{Uniform Cost Search (Graph)}
\begin{algorithmic}[1]
\Function{UNIFORM-COST-SEARCH}{problem} \textbf{returns } a solution \text{ or } failure
	\State $\textit{node} \gets  \text{a node with STATE = } problem.\text{INITIAL-STATE, PATH-COST = 0}$

	\State $\textit{frontier} \gets \text{a priority queue ordered by PATH-COST, with } node \text{ as the only element}$

	\State $\textit{explored} \gets  \text{an empty set}$
	\BState
	
	\Loop  $\textbf{ do}$
		\If{$ frontier is \emptyset$} 
			\Return $failure$
		\EndIf

		\State $\textit{node} \gets \text{POP}(frontier) $
		
		\If{$problem\text{GOAL-TEST}(node.\text{STATE}) $}
			\Return $\text{SOLUTION}(node)$
		\EndIf
		\BState

		\State $\text{add } node.\text{STATE to } explored$
		\For{$\textbf{each } action \textbf{ in } problem.\text{ACTIONS}(node.\text{STATE})$}
			\State $ child \gets \text{CHILD-NODE}(problem, node, action)$
			\If{$child.\text{STATE } \notin explored \text{ or } frontier $}
				\State $frontier\text{APPEND}(child)$
			\ElsIf{$child.\text{STATE} \in frontier \text{ with higher PATH-COST} $}
				\State $frontier.\text{REPLACE}(\text{node-with-higher-cost}, child)$
			\EndIf
		\EndFor
	\EndLoop
\EndFunction
\end{algorithmic}
\end{algorithm}
