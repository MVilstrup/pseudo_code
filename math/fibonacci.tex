		def fib(n):
		    lambda1 = (1 + sqrt(5))/2
		    lambda2 = (1 - sqrt(5))/2
		    return (lambda1**n - lambda2**n) / sqrt(5)
		def fib_approx(n)
		    # for practical range, percent error < 10^-6
		    return 1.618034**n / sqrt(5)
